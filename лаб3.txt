CREATE DATABASE lab3;
USE lab3;

CREATE TABLE STUD(
    id INT PRIMARY KEY IDENTITY(1,1),
    last_name NVARCHAR(25) NOT NULL,
    f_name NVARCHAR(25) NOT NULL,
    s_name NVARCHAR(25),
    br_date DATE,
    in_date DATE,
    exm FLOAT
);

CREATE TABLE FACULTY (
    id INT PRIMARY KEY IDENTITY(1,1),
    faculty_name NVARCHAR(50) NOT NULL
);

CREATE TABLE FORM (
    id INT PRIMARY KEY IDENTITY(1,1),
    form_name NVARCHAR(50) NOT NULL
);

CREATE TABLE HOURSS (
    id INT PRIMARY KEY IDENTITY(1,1),
    course INT NOT NULL,
    faculty_id INT NOT NULL,
    form_id INT NOT NULL,
    all_h INT,
    inclass_h INT,
    CONSTRAINT FK_hours_faculty FOREIGN KEY (faculty_id) REFERENCES faculty(id),
    CONSTRAINT FK_hours_form FOREIGN KEY (form_id) REFERENCES form(id)
);

CREATE TABLE PROCESS (
    stud_id INT NOT NULL,
    hours_id INT NOT NULL,
    CONSTRAINT PK_stud_hours PRIMARY KEY (stud_id, hours_id),
    CONSTRAINT FK_process_stud FOREIGN KEY (stud_id) REFERENCES stud(id),
    CONSTRAINT FK_process_hours FOREIGN KEY (hours_id) REFERENCES hours(id)
);

--заочная форма и младше 37
SELECT stud.last_name, stud.f_name, stud.s_name, faculty.faculty_name, hours.course FROM stud
JOIN process ON stud.id = process.stud_id
JOIN hours ON process.hours_id = hours.id
JOIN faculty ON hours.faculty_id = faculty.id
JOIN form ON hours.form_id = form.id
WHERE form.form_name = 'заочная'
AND DATEDIFF(year, stud.br_date, GETDATE()) < 37;

--количество студентов на каждом факультете 
SELECT faculty.faculty_name, COUNT(stud.id) AS student_count
FROM faculty
JOIN hours ON faculty.id = hours.faculty_id
JOIN process ON hours.id = process.hours_id
JOIN stud ON process.stud_id = stud.id
GROUP BY faculty.faculty_name;

--количество студентов, учащихся на каждой форме обучения
SELECT form.form_name, COUNT(stud.id) AS student_count
FROM form
JOIN hours ON form.id = hours.form_id
JOIN process ON hours.id = process.hours_id
JOIN stud ON process.stud_id = stud.id
GROUP BY form.form_name;

--средний возраст для каждого факультета на конец года
SELECT faculty.faculty_name, AVG(DATEDIFF(year, stud.br_date, GETDATE())) AS faculty_br FROM faculty
JOIN hours ON faculty.id = hours.faculty_id
JOIN process ON hours.id = process.hours_id
JOIN stud ON process.stud_id = stud.id
GROUP BY faculty.faculty_name;

--студенты без отчества 
SELECT stud.last_name, stud.f_name, stud.in_date, faculty.faculty_name, hours.course, form.form_name FROM stud
JOIN process ON stud.id = process.stud_id
JOIN hours ON process.hours_id = hours.id
JOIN faculty ON hours.faculty_id = faculty.id
JOIN form ON hours.form_id = form.id
WHERE stud.s_name IS NULL;

--факультет, на который поступило большее количество человек в 2015
SELECT faculty.faculty_name, COUNT(stud.id) AS stud_faculty
FROM stud
JOIN process ON stud.id = process.stud_id
JOIN hours ON process.hours_id = hours.id
JOIN faculty ON hours.faculty_id = faculty.id
WHERE YEAR(stud.in_date) = 2015
GROUP BY faculty.faculty_name
ORDER BY stud_faculty DESC;

--число студентов, поступивших  на каждый факультет в 2014 
SELECT faculty.faculty_name, form.form_name, COUNT(stud.id) AS cnt
FROM stud
JOIN process ON stud.id = process.stud_id
JOIN hours ON process.hours_id = hours.id
JOIN faculty ON hours.faculty_id = faculty.id
JOIN form ON hours.form_id = form.id
WHERE YEAR(stud.in_date) = 2014
GROUP BY faculty.faculty_name, form.form_name;

--факультеты с заочной формой обучения
SELECT DISTINCT faculty.faculty_name
FROM faculty
JOIN hours ON faculty.id = hours.faculty_id
JOIN form ON hours.form_id = form.id
WHERE form.form_name = 'заочная';

--список всех факультетов, для каждого факультета список форм обучения и номера курсов
SELECT faculty.faculty_name, form.form_name, hours.course
FROM faculty
JOIN hours ON faculty.id = hours.faculty_id
JOIN form ON hours.form_id = form.id
ORDER BY faculty.faculty_name, form.form_name, hours.course;

--список факультетов,с указанием числа студентов,обучающихся на каждой форме обучения
SELECT faculty.faculty_name, form.form_name, COUNT(stud.id) AS form_name FROM faculty
JOIN hours ON faculty.id = hours.faculty_id
JOIN form ON hours.form_id = form.id
JOIN process ON hours.id = process.hours_id
JOIN stud ON process.stud_id = stud.id
GROUP BY faculty.faculty_name, form.form_name
ORDER BY faculty.faculty_name;

--число всех студентов первого и третьего курсов для всех факультетов и форм
SELECT faculty.faculty_name, form.form_name, COUNT(stud.id) AS stud_name FROM faculty
JOIN hours ON faculty.id = hours.faculty_id
JOIN form ON hours.form_id = form.id
JOIN process ON hours.id = process.hours_id
JOIN stud ON process.stud_id = stud.id
GROUP BY faculty.faculty_name, form.form_name
ORDER BY faculty.faculty_name;

--для всех факультетов найти число иностранцев
SELECT faculty.faculty_name, form.form_name, COUNT(stud.id) AS form_id FROM stud
JOIN process ON stud.id = process.stud_id
JOIN hours ON process.hours_id = hours.id
JOIN faculty ON hours.faculty_id = faculty.id
JOIN form ON hours.form_id = form.id
WHERE hours.course IN (1, 3)
GROUP BY faculty.faculty_name, form.form_name;

--количество студентов для каждого курса с учетом факультета со средним балом выше 7,5
SELECT faculty.faculty_name, hours.course, COUNT(stud.id) AS cnt
FROM stud
JOIN process ON stud.id = process.stud_id
JOIN hours ON process.hours_id = hours.id
JOIN faculty ON hours.faculty_id = faculty.id
WHERE stud.exm > 7.5
GROUP BY faculty.faculty_name, hours.course
ORDER BY faculty.faculty_name, hours.course;

--список всех факультетов с учетом формы с указанием числа  студентов старше 45
JOIN process ON stud.id = process.stud_id
JOIN hours ON process.hours_id = hours.id
JOIN faculty ON hours.faculty_id = faculty.id
JOIN form ON hours.form_id = form.id
WHERE DATEDIFF(year, stud.br_date, GETDATE()) > 45
GROUP BY faculty.faculty_name, form.form_name
ORDER BY faculty.faculty_name;

--информация для каждого факультета, формы, курса о количестве студентов младше 27
SELECT faculty.faculty_name, form.form_name, hours.course, COUNT(stud.id) AS cnt
FROM stud
JOIN process ON stud.id = process.stud_id
JOIN hours ON process.hours_id = hours.id
JOIN faculty ON hours.faculty_id = faculty.id
JOIN form ON hours.form_id = form.id
WHERE DATEDIFF(year, stud.br_date, GETDATE()) < 27
GROUP BY faculty.faculty_name, form.form_name, hours.course
ORDER BY faculty.faculty_name, form.form_name, hours.course;

--количество студентов с фамилиями на С на каждом факультете
SELECT faculty.faculty_name, COUNT(stud.id) AS facult_id FROM stud
JOIN process ON stud.id = process.stud_id
JOIN hours ON process.hours_id = hours.id
JOIN faculty ON hours.faculty_id = faculty.id
WHERE stud.last_name LIKE 'С%'
GROUP BY faculty.faculty_name
ORDER BY faculty.faculty_name;
