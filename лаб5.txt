CREATE TABLE Workers
(
    WorkerID INT IDENTITY(1,1) PRIMARY KEY,
    Name NVARCHAR(50) NOT NULL,
    Surname NVARCHAR(50) NOT NULL,
    BirthDt DATE NOT NULL
);

CREATE TABLE Catalogs
(
    CatalogID INT IDENTITY(1,1) PRIMARY KEY,
    CatalogTitle NVARCHAR(60) NOT NULL
);

CREATE TABLE Goods
(
    GoodsID INT IDENTITY(1,1) PRIMARY KEY,
    GoodsName NVARCHAR(100) NOT NULL,
    Cost INT NOT NULL,
    CatalogID INT NOT NULL,
    CONSTRAINT FK_Goods_Catalogs
        FOREIGN KEY (CatalogID)
        REFERENCES Catalogs(CatalogID)
);

CREATE FUNCTION GetYearsOld
(
    @DateBirth DATE
)
RETURNS INT
AS
BEGIN
    DECLARE @Age INT;

    SET @Age = DATEDIFF(YEAR, @DateBirth, GETDATE());

    IF DATEADD(YEAR, @Age, @DateBirth) > GETDATE()
        SET @Age = @Age - 1;

    RETURN @Age;
END;
GO

SELECT
    Name,
    Surname,
    dbo.GetYearsOld(BirthDt) AS AgeYears
FROM Workers
WHERE dbo.GetYearsOld(BirthDt) >= 30;
GO

CREATE FUNCTION ProductsFromCatalog
(
    @CatalogName NVARCHAR(60)
)
RETURNS TABLE
AS
RETURN
(
    SELECT
        g.GoodsID,
        g.GoodsName,
        g.Cost
    FROM Goods g
    WHERE g.CatalogID IN
    (
        SELECT CatalogID
        FROM Catalogs
        WHERE CatalogTitle = @CatalogName
    )
);
GO

SELECT *
FROM dbo.ProductsFromCatalog(N'Бытовая техника');
GO

CREATE FUNCTION FormatPersonName
(
    @Name NVARCHAR(50),
    @Surname NVARCHAR(50)
)
RETURNS NVARCHAR(120)
AS
BEGIN
    DECLARE @Result NVARCHAR(120);

    SET @Result = RTRIM(@Surname) + N', ' + RTRIM(@Name);

    RETURN @Result;
END;
GO

SELECT dbo.FormatPersonName(Name, Surname) AS PersonInfo
FROM Workers;
GO

CREATE FUNCTION AvgGoodsCost
(
    @CatalogNumber INT
)
RETURNS INT
AS
BEGIN
    DECLARE @AvgPrice INT;

    SELECT @AvgPrice = CAST(AVG(Cost) AS INT)
    FROM Goods
    WHERE CatalogID = @CatalogNumber;

    RETURN ISNULL(@AvgPrice, 0);
END;
GO

SELECT
    CatalogTitle,
    dbo.AvgGoodsCost(CatalogID) AS MeanPrice
FROM Catalogs;
GO
